61  ! A = box(1), length of unit cell vector along x-axis;
62  ! B = box(2), length of unit cell vector in xy-plane;
63  ! C = box(3), length of unit cell vector in yz-plane;
64  ! alpha = box(4), cosine of angle between B and C
65  ! beta = box(5), cosine of angle between A and C
66  ! gamma = box(6), cosine of angle between A and B
67 
68  ! New matrix made up of box vectors:
69 
70  ! ax bx cx
71  ! 0 by cy
72  ! 0 0 cz
73 
74  ! convert angles to box vectors
75 
76  ! A**2 = ax**2 + ay**2 + az**2
77  ! A**2 = ax**2 + (0)**2 + (0)**2
78  ! ax = A
79  tbox(1,1) = box(1)
80 
81  ! dot(A_vec,B_vec) = ax*bx + ay*by + az*bz
82  ! dot(A_vec,B_vec) = ax*bx + (0)*by + (0)*bz
83  ! A*B*gamma = ax*bx
84  ! bx = B*gamma
85  tbox(1,2) = box(2)*box(6)
86 
87  ! dot(A_vec,C_vec) = ax*cx + ay*cy + az*cz
88  ! dot(A_vec,C_vec) = ax*cx + (0)*cy + (0)*cz
89  ! A*C*beta = ax*cx
90  ! cx = C*beta
91  tbox(1,3) = box(3)*box(5)
92 
93  ! B**2 = bx**2 + by**2 + bz**2
94  ! B**2 = bx**2 + by**2 + (0)**2
95  ! by = sqrt(B**2 - bx**2)
96  tbox(2,2) = dsqrt(box(2)**2-tbox(1,2)**2)
97 
98  ! dot(B_vec,C_vec) = bx*cx + by*cy + (0)*cz
99  ! B*C*alpha = bx*cx + by*cy
100  ! cy = (B*C*alpha - bx*cx) / by
101  tbox(2,3) = (box(2)*box(3)*box(4) - tbox(1,2)*tbox(1,3))/tbox(2,2)
102 
103  ! C = cx**2 + cy**2 + cz**2
104  ! cz = sqrt(C**2 - cx**2 - cy**2)
105  tbox(3,3) = dsqrt(box(3)**2-tbox(1,3)**2-tbox(2,3)**2)

114  pbc(1) = pbc(1) - tbox(1,3) * shift
115  end if
